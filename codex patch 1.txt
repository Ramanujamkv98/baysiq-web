 function toOrderMonth(dateStr: string): string {
   const trimmed = (dateStr ?? "").trim();
   if (!trimmed) return "";
 
   const d = new Date(trimmed);
   if (Number.isNaN(d.getTime())) return "";
 
   const y = d.getFullYear();
   const m = String(d.getMonth() + 1).padStart(2, "0");
   return `${y}-${m}`;
 }
